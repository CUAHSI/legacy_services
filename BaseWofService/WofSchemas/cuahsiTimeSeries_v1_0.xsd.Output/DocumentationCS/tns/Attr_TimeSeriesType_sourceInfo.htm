tns.ISourceInfoType SourceInfo
  Property Description  
    Property Name SourceInfo  
    Property Type tns.ISourceInfoType  
    Parent Class tns.TimeSeriesType  
    XML Type Element  
    Accessors Read/Write  
    Description A mandatory ISourceInfoType  
    Remarks It is mandatory and therefore must be populated within the XML.
The Property is initially set to null. It MUST have an object assigned to it before ToXml or ToXmlFile are called.
The object type is an interface, this means that it must be populated using a class derived from ISourceInfoType.
In order to populate it a new class deriving from tns.ISourceInfoType must be created and assigned to the property.
 
obj.SourceInfo = new XXXX();
 
Where XXXX is one of the following classes
           DatasetInfo
           DataSetInfoType
           SiteInfoType
           SourceInfoType


  
    Example 
// One of the following
tns.DatasetInfo oItem = new tns.DatasetInfo();
tns.DataSetInfoType oItem = new tns.DataSetInfoType();
tns.SiteInfoType oItem = new tns.SiteInfoType();
tns.SourceInfoType oItem = new tns.SourceInfoType();
tns.TimeSeriesType oParent = ...;

// The property MUST be populated for the XML document to be valid.
// Because the item is abstract, it can contain anyone of the 
// classes listed above, so can not be pre-populated by the generator
oParent.SourceInfo = oItem;

// Now its populated it can be used.
oParent.SourceInfo.[Some Property] = ...;

// Note:
// It may be nessarsary to cast the class to its actaul type
// (see types listed above for posibles) in order to access 
// all of its properties i.e.
if (oParent.SourceInfo[0].GetType().Name == "DatasetInfo")
{
  tns.DatasetInfo oDervItem;
  oDervItem = (tns.DatasetInfo)oParent.SourceInfo[1];
  oDervItem.[Some Property] = ...;
}
if (oParent.SourceInfo[0].GetType().Name == "DataSetInfoType")
{
  tns.DataSetInfoType oDervItem;
  oDervItem = (tns.DataSetInfoType)oParent.SourceInfo[1];
  oDervItem.[Some Property] = ...;
}
if (oParent.SourceInfo[0].GetType().Name == "SiteInfoType")
{
  tns.SiteInfoType oDervItem;
  oDervItem = (tns.SiteInfoType)oParent.SourceInfo[1];
  oDervItem.[Some Property] = ...;
}
if (oParent.SourceInfo[0].GetType().Name == "SourceInfoType")
{
  tns.SourceInfoType oDervItem;
  oDervItem = (tns.SourceInfoType)oParent.SourceInfo[1];
  oDervItem.[Some Property] = ...;
}
