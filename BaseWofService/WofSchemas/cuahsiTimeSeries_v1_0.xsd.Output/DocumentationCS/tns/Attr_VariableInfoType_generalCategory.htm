tns.Enumerations.GeneralCategoryEnum GeneralCategory
  Property Description  
    Property Name GeneralCategory  
    Property Type tns.Enumerations.GeneralCategoryEnum  
    Parent Class tns.VariableInfoType  
    XML Type Element  
    Accessors Read/Write  
    Description An optional GeneralCategoryEnum enumeration  
    Remarks It is optional, initially it is not populated.
The property IsValidGeneralCategory indicates if the property has been populated.
Any attempt to read a property that has not been populated will raise an exception.
To 'un-populate' the property SetValidGeneralCategory = false.


  
    Also See IsValidGeneralCategory - Indicates if the property has been populated.  
    Example 
GeneralCategoryEnum eItem;
tns.VariableInfoType oParent = ...;

// The property is optional so is not initially set.
// attempts to read it will result in an exception being raised
System.Diagnostics.Debug.WriteLine(oParent.GeneralCategory); // ERROR
      
// Before attempting to read the property you can test to see if ts valid
if (oParent.IsValidGeneralCategory)
{
  // its safe to read the value
  System.Diagnostics.Debug.WriteLine( oParent.GeneralCategory);
}

System.Diagnostics.Debug.Assert( oParent.IsValidGeneralCategory == false);

// Setting the value, use any one of the following
oParent.GeneralCategory = tns.Enumerations.GeneralCategoryEnum.WaterSpaceQuality;
oParent.GeneralCategory = tns.Enumerations.GeneralCategoryEnum.Climate;
oParent.GeneralCategory = tns.Enumerations.GeneralCategoryEnum.Hydrology;
oParent.GeneralCategory = tns.Enumerations.GeneralCategoryEnum.Geology;
oParent.GeneralCategory = tns.Enumerations.GeneralCategoryEnum.Biota;
oParent.GeneralCategory = tns.Enumerations.GeneralCategoryEnum.Unknown;
oParent.GeneralCategory = tns.Enumerations.GeneralCategoryEnum.Instrumentation;
System.Diagnostics.Debug.Assert( oParent.IsValidGeneralCategory == true);
