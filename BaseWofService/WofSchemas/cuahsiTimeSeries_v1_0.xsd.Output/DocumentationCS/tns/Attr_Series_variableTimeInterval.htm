tns.ITimePeriodType VariableTimeInterval
  Property Description  
    Property Name VariableTimeInterval  
    Property Type tns.ITimePeriodType  
    Parent Class tns.Series  
    XML Type Element  
    Accessors Read/Write  
    Description A mandatory ITimePeriodType  
    Remarks It is mandatory and therefore must be populated within the XML.
The Property is initially set to null. It MUST have an object assigned to it before ToXml or ToXmlFile are called.
The object type is an interface, this means that it must be populated using a class derived from ITimePeriodType.
In order to populate it a new class deriving from tns.ITimePeriodType must be created and assigned to the property.
 
obj.VariableTimeInterval = new XXXX();
 
Where XXXX is one of the following classes
           TimeIntervalType
           TimePeriodRealTimeType
           TimePeriodType
           TimeSingleType


  
    Example 
// One of the following
tns.TimeIntervalType oItem = new tns.TimeIntervalType();
tns.TimePeriodRealTimeType oItem = new tns.TimePeriodRealTimeType();
tns.TimePeriodType oItem = new tns.TimePeriodType();
tns.TimeSingleType oItem = new tns.TimeSingleType();
tns.Series oParent = ...;

// The property MUST be populated for the XML document to be valid.
// Because the item is abstract, it can contain anyone of the 
// classes listed above, so can not be pre-populated by the generator
oParent.VariableTimeInterval = oItem;

// Now its populated it can be used.
oParent.VariableTimeInterval.[Some Property] = ...;

// Note:
// It may be nessarsary to cast the class to its actaul type
// (see types listed above for posibles) in order to access 
// all of its properties i.e.
if (oParent.VariableTimeInterval[0].GetType().Name == "TimeIntervalType")
{
  tns.TimeIntervalType oDervItem;
  oDervItem = (tns.TimeIntervalType)oParent.VariableTimeInterval[1];
  oDervItem.[Some Property] = ...;
}
if (oParent.VariableTimeInterval[0].GetType().Name == "TimePeriodRealTimeType")
{
  tns.TimePeriodRealTimeType oDervItem;
  oDervItem = (tns.TimePeriodRealTimeType)oParent.VariableTimeInterval[1];
  oDervItem.[Some Property] = ...;
}
if (oParent.VariableTimeInterval[0].GetType().Name == "TimePeriodType")
{
  tns.TimePeriodType oDervItem;
  oDervItem = (tns.TimePeriodType)oParent.VariableTimeInterval[1];
  oDervItem.[Some Property] = ...;
}
if (oParent.VariableTimeInterval[0].GetType().Name == "TimeSingleType")
{
  tns.TimeSingleType oDervItem;
  oDervItem = (tns.TimeSingleType)oParent.VariableTimeInterval[1];
  oDervItem.[Some Property] = ...;
