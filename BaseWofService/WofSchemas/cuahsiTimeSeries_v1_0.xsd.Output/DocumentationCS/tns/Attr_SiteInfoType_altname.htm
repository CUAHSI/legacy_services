LiquidTechnologies.Runtime.Net20.stringCol Altname
  Property Description  
    Property Name Altname  
    Property Type LiquidTechnologies.Runtime.Net20.stringCol  
    Parent Class tns.SiteInfoType  
    XML Type Element  
    Accessors Read Only  
    Description A collection of String's  
    Remarks This collection may contain 0 to Many String's.

 
    Example 
tns.SiteInfoType oParent = ...;
 
// The collection object will already exist, but we may want to add items to it

// This adds an item to the end of the collection
oParent.Altname.Add "Sample Value";

//-------------------------------------------------------------
// To iterate through the collection

// using the iterator
foreach(String oItem in oParent.Altname)
{
  System.Diagnostics.Debug.WriteLine(oItem.ToString());
}

// using an index
for(int i=0; i<oParent.Altname.Count; i++)
{
  String oItem = oParent.Altname[i];
  System.Diagnostics.Debug.WriteLine(oItem.ToString());
}
